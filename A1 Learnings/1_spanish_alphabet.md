# Spanish Alphabet

    a             = ah  
    b (be)        = beh     
    c (ce)        = theh    
    d (de)        = deh     
    e             = eh  
    f (efe)       = eh-feh  
    g (ge)        = heh     
    h (hache)     = ah-cheh     
    i             = ee  
    j (jota)      = hoh-tah  
    k (ka)        = kah     
    l (ele)       = eh-leh  
    m (eme)       = eh-meh  
    n (ene)       = eh-neh  
    ñ (eñe)       = eh-nyeh     
    o             = oh  
    p (pe)        = peh     
    q (cu)        = kooh   
    r (erre)      = eh-ereh     
    s (ese)       = eh-seh  
    t (te)        = teh     
    u             = ooh     
    v (uve)       = ooh-veh (soft v sound)    
    w (uve doble) = ooh-veh doh-bleh (soft v sound)    
    x (equis)     = eh-kis  
    y (igriega)   = ee-gree-eh-gah  
    z (zeta)      = theh-tah
